CREATE DATABASE FLORICULTURA
USE FLORICULTURA
GO

CREATE TABLE Menu(
   ID INT NOT NULL IDENTITY,
   ITEM VARCHAR(100) NOT NULL,
   ROTA VARCHAR(100) NOT NULL,
   ORDEM INT NOT NULL,
   PRIMARY KEY (ID)
);
GO
CREATE TABLE SubMenu(
   ID INT NOT NULL IDENTITY,
   ITEM VARCHAR(100) NOT NULL,
   ROTA VARCHAR(100) NOT NULL,
   ORDEM INT NOT NULL,
   IDMENU INT,
   PRIMARY KEY (ID),
   FOREIGN KEY (IDMENU) REFERENCES Menu(ID)
);
GO
